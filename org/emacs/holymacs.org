* Emacs configuration

#+begin_src elisp
  (use-package emacs
    :ensure nil
    :defer
    :hook ((after-init . toggle-frame-maximized))
    :custom
    (dired-dwim-target t)
    (dired-listing-switches "-alth")
    ;; (completion-auto-help nil)
    (enable-recursive-minibuffers t)
    (initial-buffer-choice #'eshell)
    (make-backup-files nil)
    (shr-inhibit-images t)
    (tab-always-indent 'complete)
    (use-short-answers t)
    (custom-file "~/.emacs.d.holymacs/custom.el")
    (completion-styles '(basic flex))
    :config
    (tool-bar-mode -1)
    (menu-bar-mode -1)
    (scroll-bar-mode -1)
    (savehist-mode 1)
    (recentf-mode 1)
    (ffap-bindings)
    (repeat-mode 1)
    :bind
    ("s-k" . #'execute-extended-command)
    ("s-j" . #'switch-to-buffer)
    ("s-0" . #'delete-window)
    ("s-1" . #'delete-other-windows)
    ("s-2" . #'split-window-below)
    ("s-3" . #'split-window-right)
    ("<XF86AudioLowerVolume>" . #'pulseaudio-control-decrease-sink-volume)
    ("<XF86AudioRaiseVolume>" . #'pulseaudio-control-increase-sink-volume)
    ("<XF86AudioMute>" . #'pulseaudio-control-toggle-current-sink-mute))

  (if (file-exists-p custom-file) (load-file custom-file))
#+end_src

#+begin_src elisp
  (use-package exwm
    :init
    (async-shell-command "dex -a -d -e exwm" nil nil);"*dex-errors*")
    :config
    (server-mode 1)
    (require 'exwm)
    (setq exwm-workspace-number 4)
    ;; Make class name the buffer name.
    (add-hook 'exwm-update-class-hook
	      (lambda () (exwm-workspace-rename-buffer exwm-class-name)))
    ;; Global keybindings.
    (setq exwm-systemtray-height 30
	  exwm-input-global-keys
	  `(([?\s-k] . execute-extended-command)
	    ([?\s-j] . switch-to-buffer)
	    ([?\s-n] . holymacs-mode/body)))
    (exwm-enable)
    (require 'exwm-systemtray)
    (exwm-systemtray-mode 1))
#+end_src
