Some useful functions integrating =vterm= and *Tmux*.

#+begin_src elisp
  (defun holymacs-tmux/open ()
    (interactive)
    (holymacs-vterm-exec "tmux attach && exit"))

  (defun holymacs-vterm-exec (command)
    (interactive "sCommand: ")
    (let ((buffer-name (format "*vterm <%s>*" command)))
      (with-current-buffer (vterm buffer-name)
	(vterm-send-string (format "%s\n" command)))))
#+end_src
