** Generate site

Generate your static blog using [[https://orgmode.org/manual/Publishing.html][Org-mode publishing]].

#+name: site-publish
#+begin_src emacs-lisp :results output none
  (defun mm/blog-setup ()
    "Setup for my blog"
    (org-publish-project-alist
     '(
       ("site"
	:base-directory "./"
	:base-extension "org"
	:publishing-directory "~/site"
	:recursive t
	:publishing-function org-html-publish-to-html
	:export-babel-evaluate nil
	:headline-levels 4
	:with-toc nil)
       ("assets"
	:base-directory "./assets/"
	:base-extension "css\\|js\\|png\\|jpg\\|gif\\|pdf\\|mp3\\|ogg\\|swf"
	:publishing-directory "~/site/assets/"
	:recursive t
	:publishing-function org-publish-attachment)
       ("org" :components ("site" "assets"))
       )))
    #+end_src

    Then you can invoke ~mm/blog-setup~.
    
** Autopublishing script.


#+begin_src shell :var t=(current-time-string) :tangle no :results output none
cd .build
git add .

git push -f origin master

#+end_src

#+begin_src emacs-lisp :results output none
  (let* ((remote "git@github:MinasMazar/minasmazar.github.io.git")
	(commit-msg (concat "'Autopublish [" (current-time-string) "]'"))
	(cd "~/site")
	(out (shell-command-to-string "git remote"))
	(repo-p (string-match "fatal" out)))
    (unless repo-p (progn 
		     (shell-command "git init")
		     (shell-command (concat "git remote add origin " remote))))
    (shell-command "git add .")
    (shell-command (concat "git commit -m " commit-msg))
    (shell-command "git push origin master"))
#+end_src
